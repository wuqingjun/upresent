Wildcard Matching 

class Solution {
public:
    bool isMatch(const char *s, const char *p) 
    {
        for(const char * star = NULL, *laststarmatch = NULL; *s; )
        {
            if('?' == *p || *p == *s) {++s, ++p;}
            else if('*' == *p) {star = p++, laststarmatch = s;}
            else if(star) { p = star + 1, s = ++laststarmatch; }
            else {return false;}
        }
        while('*' == *p) {++p;}
        return !*p;
    }
};
