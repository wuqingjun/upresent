Word Ladder II 

class Solution {
public:
    vector<vector<string>> findLadders(string start, string end, unordered_set<string> &dict)
    {
        unordered_map<string, vector<string> > adj;
        unordered_set<string> que[2];
        que[0].insert(start);
        unordered_map<string, int> visited;
        visited[start] = 0;

        bool found = false;
        int i = 0;
        for (; !found && !que[i % 2].empty(); ++i)
        {
            for (string s; !que[i % 2].empty(); que[i % 2].erase(s))
            {
                s = *que[i % 2].begin();
                if (s == end)
                {
                    found = true;
                }
                for (int n = 0; n < s.size(); ++n)
                {
                    string t = s;
                    for (int k = 0; k < 26; ++k)
                    {
                        t[n] = 'a' + k;
                        if (dict.find(t) != dict.end() && (visited.find(t) == visited.end() || visited[t] >= i))
                        {
                            adj[t].push_back(s);
                            que[(i + 1) % 2].insert(t);
                            visited[t] = i;
                        }
                    }
                }
            }
        }
        vector<vector<string> > res;
        vector<string> path(1, end);
        bt(path, adj, res, start, i);
        return res;
    }

    void bt(vector<string> &path, unordered_map<string, vector<string> > &adj, vector<vector<string> > &res, string &start, int currLen)
    {
        if (path.front() == start)
        {
            res.push_back(path);
        }
        else
        {
            for (auto & e : adj[path.front()])
            {
                path.insert(begin(path), e);

                bt(path, adj, res, start, currLen - 1);

                path.erase(begin(path));
            }
        }
    }
};
