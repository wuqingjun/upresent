Reverse Integer 

class Solution {
public:
    int reverse(int x) 
    {
        bool negative = x < 0;
        long long res = 0;
        while(x)
        {
            res = res * 10 + x % 10;
            x /= 10;
        }
        if(!negative && res > INT_MAX) return 0;
        if(negative && res < INT_MIN) return 0;
        return res;
    }
};
