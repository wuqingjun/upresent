Longest Valid Parentheses 

class Solution {
public:
    int longestValidParentheses(string s) 
    {
        int res = scan(s, '(');
        reverse(begin(s), end(s));
        return max(res, scan(s,')'));
    }
    int scan(string & s, char c)
    {
        int res = 0;
        for(int i = 0, l = 0, count = 0; i <s.length(); ++i)
        {
            l += s[i] == c ? 1 : -1, ++count;
            if(l == 0) { res = max(res, count); }
            if(l < 0) { count = 0, l = 0; }
        }
        return res;
    }
};
