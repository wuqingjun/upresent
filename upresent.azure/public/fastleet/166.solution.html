Fraction to Recurring Decimal 

class Solution {
public:
    string fractionToDecimal(long long numerator, long long denominator)
    {
        bool flag = !(numerator * denominator < 0);
        numerator = abs(numerator), denominator = abs(denominator);
        char buff[32] = { 0 };
        sprintf(buff, "%lld", numerator / denominator);
        string res(buff);
        unordered_map<long long, size_t> mp;
        bool isDot = true;
        for (numerator %= denominator; numerator != 0; numerator %= denominator, numerator *= 10)
        {
            res.append(1, isDot ? '.' : ('0' + numerator / denominator));
            isDot = false;
            if (mp.find(numerator % denominator) == end(mp))
            {
                mp[numerator % denominator] = res.length();
            }
            else
            {
                res = res.substr(0, mp[numerator % denominator]) + "(" + res.substr(mp[numerator % denominator]) + ")";
                numerator = 0;
            }
        }
        return flag ? res : "-" + res;
    }
};
