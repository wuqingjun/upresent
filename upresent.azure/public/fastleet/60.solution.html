Permutation Sequence 

class Solution {
public:
    string getPermutation(int n, int k) 
    {
        int factorials[9] = {1, 1, 2, 6, 24, 120, 720, 5040, 40320};
        string s("123456789"), res;
        for(int i = n - 1; i >= 0; --i)
        {
            n = k / factorials[i], k %= factorials[i];
            if(k == 0){ --n, k = factorials[i]; }
            res.append(1, s[n]), s.erase(n, 1);
        }
        return res;
    }
};
