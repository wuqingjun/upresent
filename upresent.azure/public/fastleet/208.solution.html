Implement Trie (Prefix Tree) 

class TrieNode
{
public:
    // Initialize your data structure here.
    TrieNode(char c) : ch(c), end(false) { }
public:
    char ch;
    bool end;
    unordered_map<char, TrieNode *> nodes;
};

class Trie {
public:
    Trie() : root(new TrieNode('\0')) { }

    // Inserts a word into the trie.
    void insert(string s)
    {
        int i = 0;
        TrieNode *p = helper(s, i);
        for (; i < s.size(); ++i)
        {
            auto t = new TrieNode(s[i]);
            p->nodes[s[i]] = t;
            p = t;
        }
        p->end = true;
    }

    // Returns if the word is in the trie.
    bool search(string key)
    {
        int i = 0;
        TrieNode *p = helper(key, i);
        return i == key.size() && p->end;
    }

    // Returns if there is any word in the trie
    // that starts with the given prefix.
    bool startsWith(string prefix)
    {
        int i = 0;
        TrieNode *p = helper(prefix, i);
        return i == prefix.size();
    }

private:
    TrieNode * helper(string s, int &i)
    {
        i = 0;
        TrieNode *p = root;
        for (; i < s.size() && p->nodes.find(s[i]) != end(p->nodes); ++i)
        {
            p = p->nodes[s[i]];
        }
        return p;
    }

private:
    TrieNode* root;
};
