Scramble String 

class Solution {
public:
    bool isScramble(string s1, string s2) 
    {
        int l = s1.size();
        vector<vector<vector<int>>> dp(l, vector<vector<int> >(l, vector<int>(l + 1, -1)));
        return rec(s1, 0, s2, 0, s1.size(), dp);
    }
    bool rec(string &s1, int i, string &s2, int j, int l, vector<vector<vector<int>>> &dp)
    {
        int &res = dp[i][j][l];
        if(res == -1)
        {
            res = compare(s1, i, s2, j, l);
            for(int a = 1; !res && a < l; ++a)
            {
                res = (rec(s1, i, s2, j, a, dp) && rec(s1, i + a, s2, j + a, l - a, dp)) ||
                    (rec(s1, i, s2, j + l - a, a, dp) && rec(s1, i + a, s2, j, l - a, dp));
            }
        }
        return res;
    }
    int compare(string &s1, int i, string &s2, int j, int l)
    {
        int res = 1;
        for(int a = 0; a < l && res; ++a)
        {
            res = s1[i + a] == s2[j + a];
        }
        return res;
    }
};
