Substring with Concatenation of All Words 

class Solution {
public:
    vector<int> findSubstring(string S, vector<string> &L)
    {
        vector<int> res;
        unordered_map<string, int> mp, match;
        for (auto & e : L) { mp[e]++; }
        for (int n = 0, wl = L[0].size(); n < wl; ++n, match.clear())
        {
            for (int i = n, start = 0, total = 0; i < S.size(); i += wl)
            {
                auto t = S.substr(i, wl);
                if (mp.count(t))
                {
                    match[t]++;
                    start = ++total == 1 ? i : start;
                    for (auto old = S.substr(start, wl); match[t] > mp[t]; 
                        --match[old], start += wl, old = S.substr(start, wl), --total)  {   }
                    if (total == L.size())  { res.push_back(start); }
                }
                else
                {
                    match.clear(), total = 0;
                }
            }
        }
        return res;
    }
};
