Spiral Matrix II 

class Solution {
public:
    vector<vector<int> > generateMatrix(int n)
    {
        vector<vector<int>> matrix(n, vector<int>(n, 0));
        --n;
        for (int k = 0, l = 0, m = n, e = 1; k <= m && l <= n;)
        {
            for (int i = l; i <= n; ++i)
            {
                matrix[k][i] = e++;
            }
            ++k;
            for (int i = k; i <= m; ++i)
            {
                matrix[i][n] = e++;
            }
            --n;
            for (int i = n; i >= l && k <= m; --i)
            {
                matrix[m][i] = e++;
            }
            --m;
            for (int i = m; i >= k && l <= n; --i)
            {
                matrix[i][l] = e++;
            }
            ++l;
        }
        return matrix;
    }
};
