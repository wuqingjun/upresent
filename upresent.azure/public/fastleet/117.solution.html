Populating Next Right Pointers in Each Node II 

/**
 * Definition for binary tree with next pointer.
 * struct TreeLinkNode {
 *  int val;
 *  TreeLinkNode *left, *right, *next;
 *  TreeLinkNode(int x) : val(x), left(NULL), right(NULL), next(NULL) {}
 * };
 */
class Solution {
public:
    TreeLinkNode * nextChild(TreeLinkNode * p)
    {
        for(;p; p = p->next)
        {
            if(p->left)
            {
                return p->left;
            }
            else if(p->right)
            {
                return p->right;
            }
        }
        return NULL;
    }
    void connect(TreeLinkNode *root) 
    {
        if(root)
        {
            root->next = NULL;
            rec(root);
        }
    }
    void rec(TreeLinkNode * p)
    {
        if(p)
        {
            if(p->left)
            {
                if(p->right)
                {
                    p->left->next = p->right;
                }
                else
                {
                    p->left->next = nextChild(p->next);
                }
            }
            if(p->right)
            {
                p->right->next = nextChild(p->next);
            }
            rec(p->right);
            rec(p->left);
        }
    }
};
