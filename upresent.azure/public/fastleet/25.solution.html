Reverse Nodes in k-Group 

class Solution
{
public:
    ListNode *reverseKGroup(ListNode *head, int k)
    {
        ListNode *nhead = head, *first = NULL, *prev = NULL, *next = NULL;
        for (int i = 0; head; ++i, head = next)
        {
            next = head->next;
            if (i % k == 0) { first = head; }
            if (i % k == (k - 1))
            {
                for (ListNode * curr = first, *currnext = curr->next, *prev = next; curr != next;)
                {
                    curr->next = prev, prev = curr, curr = currnext, currnext = curr ? curr->next : NULL;
                }               
                if (prev) { prev->next = head; }
                else { nhead = head; }
                prev = first, head = next;
            }
        }
        return nhead;
    }
};
