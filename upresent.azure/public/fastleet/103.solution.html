Binary Tree Zigzag Level Order Traversal 

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) 
    {
        vector<vector<int> > res;
        if(root)
        {
            deque<TreeNode *> dq[2];
            dq[0].push_front(root);
            for(int i = 0; !dq[i].empty();i = (i + 1) % 2)
            {
                vector<int> row;
                while(!dq[i].empty())
                {
                    if(i && dq[i].back())
                    {
                        row.push_back(dq[i].back()->val);
                        if(dq[i].back()->right) dq[(i + 1) % 2].push_front(dq[i].back()->right);
                        if(dq[i].back()->left) dq[(i + 1) % 2].push_front(dq[i].back()->left);
                        dq[i].pop_back();
                    }
                    if(!i && dq[i].front())
                    {
                        row.push_back(dq[i].front()->val);
                        if(dq[i].front()->left) dq[(i + 1) % 2].push_back(dq[i].front()->left);
                        if(dq[i].front()->right) dq[(i + 1) % 2].push_back(dq[i].front()->right);
                        dq[i].pop_front();
                    }
                }
                res.push_back(row);
            }
        }
        return res;
    }
};
