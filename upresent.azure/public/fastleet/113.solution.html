Path Sum II 

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> pathSum(TreeNode* root, int sum) 
    {
        vector<vector<int> > res;
        if(root)
        {
            vector<int> tempSet;
            int tempSum = 0;
            rec(root, sum, tempSum, tempSet, res);
        }
        return res;
    }
    void rec(TreeNode *root, int sum, int tempSum, vector<int> &tempSet, vector<vector<int> > & res)
    {
        tempSet.push_back(root->val);
        tempSum += root->val;
        if(!root->left && !root->right && tempSum == sum)
        {
            res.push_back(tempSet);
        }
        if(root->left)
        {
            rec(root->left, sum, tempSum, tempSet, res);
        }
        if(root->right)
        {
            rec(root->right, sum, tempSum, tempSet, res);
        }
        tempSum -= root->val;
        tempSet.pop_back();
    }
};
