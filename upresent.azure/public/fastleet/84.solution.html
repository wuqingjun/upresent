Largest Rectangle in Histogram 

class Solution {
public:
    int largestRectangleArea(vector<int> &height) 
    {
        int res = 0;
        height.push_back(0);
        stack<int> st;
        for(int i = 0; i < height.size(); ++i)
        {
            if(st.empty() || height[st.top()] < height[i])
            {
                st.push(i);
            }
            else
            {
                auto t = st.top();
                st.pop();
                res = max(res, (st.empty() ? i : (i - st.top() - 1)) * height[t]); 
                --i;
            }
        }
        return res;
    }
};
