Balanced Binary Tree 

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution 
{
private:
    int ABS(int n) {return n >= 0 ? n : -n;}
public:
    bool isBalanced(TreeNode *root) 
    {
        bool flag = true;
        height(root, flag);
        return flag;
    }
    int height(TreeNode * p, bool &flag)
    {
        if(p && flag)
        {
            int l = height(p->left, flag);
            int r = height(p->right, flag);
            if(ABS(l - r) > 1)
            {
                flag = false;
            }
            return max(l, r) + 1;
        }
        return 0;
    }
};
