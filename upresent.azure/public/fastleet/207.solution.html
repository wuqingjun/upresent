class Solution {
public:
    bool canFinish(int n, vector<vector<int>>& p)
    {
        unordered_map<int, unordered_set<int> > mp;
        for (int i = 0; i < p.size(); ++i)
        {
            mp[p[i][0]].insert(p[i][1]);
        }
        unordered_set<int> total;
        for (int i = 0; i < n; ++i)
        {
            unordered_set<int> s;
            if (!rec(i, mp, s, total))
            {
                return false;
            }
        }
        return true;
    }

    bool rec(int i, unordered_map<int, unordered_set<int> > & mp, unordered_set<int> &s, unordered_set<int> &total)
    {
        if (s.find(i) != end(s))
        {
            return false;
        }
        if (total.find(i) != end(total))
        {
            return true;
        }
        total.insert(i);
        s.insert(i);
        for (auto & e : mp[i])
        {
            if (!rec(e, mp, s, total))
            {
                return false;
            }
        }
        return true;
    }
};
