class Solution {
public:
    vector<Interval> insert(vector<Interval> &intervals, Interval ni)
    {
        vector<Interval> res;
        auto s = search(intervals, 0, intervals.size() - 1, ni.start);
        auto e = search(intervals, 0, intervals.size() - 1, ni.end);
        for (int i = 0; i < s.start; ++i) { res.push_back(intervals[i]); }
        res.push_back(Interval(s.start > s.end ? ni.start : intervals[s.start].start, e.start > e.end ? ni.end : intervals[e.start].end));
        for (int i = (e.start <= e.end ? e.start + 1 : e.start); i < intervals.size(); ++i)
        {
            res.push_back(intervals[i]);
        }
        return res;
    }
    Interval search(vector<Interval> &intervals, int l, int h, int n)
    {
        if (l <= h)
        {
            int m = (h + l) / 2;
            if (intervals[m].start <= n && intervals[m].end >= n) { return Interval(m, m); }
            return intervals[m].start > n ? search(intervals, l, m - 1, n) : search(intervals, m + 1, h, n);
        }
        return Interval(l, h);
    }
};
