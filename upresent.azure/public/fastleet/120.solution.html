Triangle 

class Solution {
public:
    int minimumTotal(vector<vector<int> > &triangle) 
    {
        vector<pair<int,int> > dp(triangle.size(), {-1, 0});
        return rec(triangle, 0, triangle.size(), 0, dp);
    }
    int rec(vector<vector<int> > & t, int row, int R, int i, vector<pair<int, int> > &dp)
    {
        if(row < R)
        {
            if(dp[row].first < i)
            {
                int l = rec(t, row + 1, R, i, dp);
                int r = rec(t, row + 1, R, i + 1, dp);
                int res = min(l, r) + t[row][i]; 
                dp[row] = {i, res};
            }
            return dp[row].second;
        }
        return 0;
    }
};
