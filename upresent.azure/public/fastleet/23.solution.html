Merge k Sorted Lists 

class Solution {
public:
    ListNode *mergeKLists(vector<ListNode *> &lists) 
    {
        int size = lists.size();
        if(size < 3) { return merge(size > 0 ? lists[0] : NULL, size > 1 ? lists[1] : NULL); }
        vector<ListNode *> temp[2];
        for(int i = 0; i < size; ++i)
        {
            temp[i % 2].push_back(lists[i]);
        }
        return merge(mergeKLists(temp[0]), mergeKLists(temp[1]) );
    }
    ListNode *merge(ListNode * l1, ListNode *l2)
    {
        ListNode * head = NULL;
        for(ListNode *curr = NULL, *temp = NULL; l1 || l2; curr = temp)
        {
            if(!l2 || l1 && l1->val < l2->val) {temp = l1, l1 = l1->next;}
            else {temp = l2, l2 = l2->next;}
            if(curr) { curr->next = temp; }
            else { head = temp; }
        }
        return head;
    }
};
