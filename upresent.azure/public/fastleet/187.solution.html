Repeated DNA Sequences 

class Solution {
public:
    vector<string> findRepeatedDnaSequences(string s) 
    {
        vector<string> res;
        unordered_map<size_t, int> hs;
        for(size_t i = 0, hn = 0; i < s.size(); ++i)
        {
            if(hs[hn = ((hn << 3) | (s[i] & 7)) & 0X03FFFFFFF]++ == 1)
            {
                res.push_back(s.substr(i - 9, 10));
            }
        }
        return res;
    }
};
