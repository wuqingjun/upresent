Longest Common Prefix 

class Solution {
public:
    string longestCommonPrefix(vector<string> &strs) 
    {
         string res;
         for(int k = 0; strs.size(); ++k)
         {
             int i = 0;
             for(; i < strs.size() && k < strs[i].size() && strs[i][k] == strs[0][k]; ++i) {}
             if(i == strs.size()) {res.append(1, strs[0][k]);}
             else {return res;}
         }
         return res;
    }
};
