Shortest Palindrome 

class Solution {
public:
    string shortestPalindrome(string s)
    {
        string t;
        for (int i = 0; i < s.size(); ++i)
        {
            t.append(1, s[i]);
            if (i < s.size() - 1)
            {
                t.append(1, '#');
            }
        }
        vector<int> r(t.size(), 1);
        int center = 0, mx = 0, le = 0;
        for (int i = 1; i < t.size(); ++i)
        {
            r[i] = 2 * center >= i ? min(r[2 * center - i], mx - i + 1) : 1;
            while (i + r[i] >= mx && i + r[i] < t.size() && i >= r[i] && t[i + r[i]] == t[i - r[i]])
            {
                ++r[i];
            }
            if (i + r[i] - 1 > mx)
            {
                mx = i + r[i] - 1, center = i;
            }
            if (i - r[i] + 1 == 0)
            {
                le = i + r[i] - 1;
            }
        }
        string res;
        for (int i = t.size() - 1; i > le; --i)
        {
            if (t[i] != '#')
            {
                res.append(1, t[i]);
            }
        }
        return res + s;
    }
};
