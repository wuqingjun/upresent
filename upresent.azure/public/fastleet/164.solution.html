Maximum Gap 

class Solution {
public:
    int maximumGap(vector<int> &num)
    {
        int n = num.size(), res = 0;
        if (n > 1)
        {
            int ma = INT_MIN, mi = INT_MAX;
            for (auto & e : num)
            {
                ma = max(e, ma);
                mi = min(e, mi);
            }

            int bucketsize = (ma - mi) <= n ? 1 : (ma - mi) / n;
            int numbuckets = (ma - mi) / bucketsize + 1;
            vector<pair<int, int> > buckets(numbuckets, { INT_MAX, INT_MIN });
            for (auto &e : num)
            {
                int i = (e - mi) / bucketsize;
                buckets[i].first = min(buckets[i].first, e);
                buckets[i].second = max(buckets[i].second, e);
            }

            for (int i = 1, prev = 0; i < numbuckets; ++i)
            {
                if (buckets[i].first != INT_MAX)
                {
                    res = max(res, buckets[i].first - buckets[prev].second);
                    prev = i;
                }
            }
        }
        return res;
    }
};
