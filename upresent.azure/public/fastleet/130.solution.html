Surrounded Regions 

class Solution {
public:
    void solve(vector<vector<char>> &board) 
    {
        int M = board.size();
        if(M)
        {
            int N = board[0].size();
            for(int i = 0; i < M; ++i)
            {
                for(int j = 0; j < N; ++j)
                {
                    bool res = true; 
                    rec(board, i, j, M, N, res);
                    if(!res)
                    {
                        paint(board, i, j, M, N, 'P', 'Q'); 
                    }
                    else
                    {
                        paint(board, i, j, M, N, 'P', 'X'); 
                    }
                }
            }
            for(int i = 0; i < M; ++i)
            {
                for(int j = 0; j < N; ++j)
                {
                    if(board[i][j] == 'Q')
                    {
                        board[i][j] = 'O';
                    }
                }
            }
        }
    }
    
    void rec(vector<vector<char> > &board, int i, int j, int I, int J, bool &flag)
    {
        if(i != -1 && i != I && j != -1 && j != J)
        {
            if(board[i][j] == 'O')
            {
                board[i][j] = 'P';
                rec(board, i - 1, j, I, J, flag);
                rec(board, i + 1, j, I, J, flag);
                rec(board, i, j - 1, I, J, flag);
                rec(board, i, j + 1, I, J, flag);
            }
        }
        else
        {
            flag = false;
        }
    }
    bool paint(vector<vector<char> > &board, int i, int j, int I, int J, char old, char n)
    {
        if(i != -1 && i != I && j != -1 && j != J)
        {
            if(board[i][j] == old)
            {
                board[i][j] = n;
                paint(board, i - 1, j, I, J, old, n); 
                paint(board, i + 1, j, I, J, old, n); 
                paint(board, i, j - 1, I, J, old, n);
                paint(board, i, j + 1, I, J, old, n);
            }
        }
    }
};
