Max Points on a Line 

/**
 * Definition for a point.
 * struct Point {
 *     int x;
 *     int y;
 *     Point() : x(0), y(0) {}
 *     Point(int a, int b) : x(a), y(b) {}
 * };
 */
class Solution {
public:
    int maxPoints(vector<Point> &points) 
    {
        int size = points.size(), res = 0;
        for(int i = 0; i < size; ++i)
        {
            int m = 0, dup = 1;
            unordered_map<int, int> mp;
            unordered_map<double, int> mp2;
            for(int j = i + 1; j < size; ++j)
            {
                if(points[i].x == points[j].x)
                {
                    if(points[i].y == points[j].y)
                    {
                        ++dup;
                    }
                    else
                    {
                        mp[points[i].x]++;
                        m = max(m, mp[points[i].x]);
                    }
                }
                else
                {
                    mp2[double(points[i].y - points[j].y) / double(points[i].x - points[j].x)]++;
                    m = max(m, mp2[double(points[i].y - points[j].y) / double(points[i].x - points[j].x)]);
                }
            }
            res = max(res, m + dup);
        }
        return res;
    }
};
