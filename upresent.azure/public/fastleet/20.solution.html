Valid Parentheses 

class Solution {
public:
    bool isValid(string s) 
    {
        stack<int> st;
        unordered_map<char, int> mp;
        string brackets("(){}[]");
        for(int i = 0; i < brackets.size(); ++i)
        {
            mp[brackets[i]] = i;
        }
        for(int i = 0; i < s.size(); ++i)
        {
            if(!st.empty() && (mp[st.top()] % 2 == 0 && mp[s[i]] == mp[st.top()] + 1 ))
            {
                st.pop();
            }
            else
            {
                st.push(s[i]);
            }
        }
        return st.empty();
    }
};
