Add and Search Word - Data structure design 

class TrieNode
{
public:
    TrieNode(char c): ch(c), end(false) {}
public:
    char ch;
    bool end;
    unordered_map<char, TrieNode *> children;
};

class WordDictionary {
public:

    // Adds a word into the data structure.
    void addWord(string word) 
    {
        TrieNode *p = root;
        int i = 0;
        for(; i < word.size() && p->children.find(word[i]) != end(p->children); ++i)
        {
            p = p->children[word[i]];
        }
        for(; i < word.size(); ++i)
        {
            p->children[word[i]] = new TrieNode(word[i]);
            p = p->children[word[i]];
        }
        p->end = true;
    }

    // Returns if the word is in the data structure. A word could
    // contain the dot character '.' to represent any one letter.
    bool search(string word) 
    {
        return rec(root, word, 0);
    }
    
public:
    WordDictionary() : root(new TrieNode('\0')) {}
private:
    TrieNode *root;
    bool rec(TrieNode * p, string &word, int i)
    {
        if(i < word.size())
        {
            if(word[i] == '.')
            {
                for(auto & e: p->children)
                {
                    if(rec(e.second, word, i + 1))
                    {
                        return true;   
                    }
                }
            }
            else if(p->children.find(word[i]) != end(p->children))
            {
                return rec(p->children[word[i]], word, i + 1);
            }
        }
        else
        {
            return p->end;
        }
        return false;
    }
};
