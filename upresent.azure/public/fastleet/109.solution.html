Convert Sorted List to Binary Search Tree 

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution 
{
private:
    vector<ListNode *> split(ListNode * l)
    {
        vector<ListNode *> res(3);
        
        if(l)
        {
            ListNode *p1 = l, *p2 = l->next, *prev = NULL;
            for(int i = 0; p2 && p2->next ; ++i, p2 = p2->next)
            {
                if(i % 2 == 0)
                {
                    prev = p1;
                    p1 = p1->next;
                }
            }
            res[0] = p1 == l ? NULL : l;
            res[1] = p1;
            res[2] = p1->next;
            if(prev)
            {
                prev->next = NULL;
            }
            p1->next = NULL;
        }
        return res;
    }
public:
    TreeNode *sortedListToBST(ListNode *head) 
    {
        auto res = split(head);
        if(res[1])
        {
            auto p = new TreeNode(res[1]->val);
            p->left = sortedListToBST(res[0]);
            p->right = sortedListToBST(res[2]);
            return p;
        }
        return NULL;
    }
};
