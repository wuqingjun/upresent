Group Anagrams 

class Solution {
public:
    vector<string> anagrams(vector<string> &strs) 
    {
        vector<string> res;
        unordered_map<string, vector<string> > mp;
        for(auto s : strs)
        {
            mp[count(s)].push_back(s);
        }
        for(auto e: mp)
        {
            if(e.second.size() > 1)
            {
                res.insert(end(res), begin(e.second), end(e.second));
            }
        }
        return res;
    }
    
    string count(string & s)
    {
        string res;
        vector<int> bm(256, 0);
        for(auto & e: s) { bm[e]++; }
        for(int i = 0; i < bm.size(); ++i)
        {
            if(bm[i] > 0)
            {
                res += string(1, char(i)) + ":" + numbertostring(bm[i]) + "|";
            }
        }
        return res;
    }
    
    string numbertostring(int n)
    {
        char buff[16] = {'\0'};
        sprintf(buff, "%d", n);
        return string(buff);
    }

};
