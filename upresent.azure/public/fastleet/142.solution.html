Linked List Cycle II 

class Solution {
public:
    bool containsNearbyAlmostDuplicate(vector<int>& nums, int k, int t) 
    {
        vector<int> temp;
        for(int i = 0; i < nums.size(); ++i)
        {
            int n = search(temp, nums[i]);
            for(int m = 0; m < 2; ++m)
            {
                if(n - m >= 0 && n - m < temp.size())
                {
                    long long ll = temp[n - m];
                    if(abs(ll - nums[i]) <= t) { return true; }
                }
            }
            temp.insert(begin(temp) + n, nums[i]);
            if(i >= k)
            {
                n = search(temp, nums[i - k]);
                temp.erase(begin(temp) + n);
            }
        }
        return false;
    }
    int search(vector<int> &nums, int t)
    {
        int l = 0, h = nums.size() - 1;
        while(l <= h)
        {
            int m = (l + h) / 2;
            if(nums[m] == t) { return m; }
            else if(nums[m] < t) { l = m + 1; }
            else { h = m - 1; }
        }
        return l;
    }
};
