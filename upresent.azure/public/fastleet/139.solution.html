Word Break 

class Solution 
{
public:
    bool wordBreak(string s, unordered_set<string> &dict) 
    {
        int N = s.size() + 1;
        vector<int> dp(N, -1);
        dp[N - 1] = 1;
        return rec(s, 0, N, dp, dict);
    }
    
    int rec(string s, int i, int N, vector<int> &dp, unordered_set<string> &dict)
    {
        if(-1 == dp[i])
        {
            dp[i] = 0;
            for(int n = 1; n < N - i && dp[i] == 0; ++n)
            {
                string s1 = s.substr(i, n);
                if(dict.find(s1) != end(dict) && rec(s, i + n, N, dp, dict))
                {
                    dp[i] = 1;
                }
            }
        }
        return dp[i];
    }
};
