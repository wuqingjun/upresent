Flatten Binary Tree to Linked List 

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    void flatten(TreeNode *root) 
    {
        TreeNode * prev = NULL;
        rec(root, prev);
    }
    void rec(TreeNode * p, TreeNode * & prev)
    {
        if(p)
        {
            if(prev)
            {
                prev->right = p;
            }
            prev = p;
            auto r = p->right;
            rec(p->left, prev);
            p->left = NULL;
            rec(r, prev);
        }
    }
};
