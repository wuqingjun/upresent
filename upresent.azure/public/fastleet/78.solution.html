Subsets 

class Solution {
public:
    vector<vector<int> > subsets(vector<int> &S) 
    {
        sort(begin(S), end(S));
        vector<vector<int> > res(1, vector<int>());
        for(int n = 0; n < S.size(); ++n)
        {
            int size = res.size();
            for(int i = 0; i < size; ++i)
            {
                vector<int> temp(res[i]);
                temp.push_back(S[n]);
                res.push_back(temp);
            }
        }
        return res;
    }
};
