Minimum Depth of Binary Tree 

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int minDepth(TreeNode *root) 
    {
        if(root)
        {
            vector<pair<TreeNode *, int> > stack(1, {root, 1});
            int depth = INT_MAX;
            while(!stack.empty())
            {
                auto b = stack.back();
                stack.pop_back();
                if(!b.first->left && !b.first->right)
                {
                    depth = min(depth, b.second);
                }
                if(b.first->left)
                {
                    stack.push_back({b.first->left, b.second + 1});
                }
                if(b.first->right)
                {
                    stack.push_back({b.first->right, b.second + 1});
                }
            }
            return depth;
        }
        return 0;
    }
};
