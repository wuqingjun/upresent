Regular Expression Matching 

class Solution {
public:
    int maxArea(vector<int> &height) 
    {
        int res = 0;
        for(int i = 0, j = height.size() - 1; i < j; )
        {
            int a = height[i], b = height[j];
            res = max(res, min(a, b) * (j - i));
            if(a <= b) { ++i; }
            if(a >= b) { --j; }
        }
        return res;
    }
};
