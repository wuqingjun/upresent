Search for a Range 

class Solution {
public:
    vector<int> searchRange(int A[], int l, int t) 
    {
        vector<int> res(2, -1);
        for(int n = 0; n < 2; ++n)
        {
            for(int i = 0, j = l - 1, m = (i + j) / 2; i <= j && res[n] != m; m = (i + j) / 2)
            {
                if(A[m] < t || (A[m] == t && n == 1 && m < j && A[m] == A[m + 1])) { i = m + 1; }
                else if(A[m] > t || (A[m] == t && n == 0 && m > 0 && A[m] == A[m - 1])) { j = m - 1; }
                else { res[n] = m; }
            }
        }
        return res;
    }
};
