class TrieNode
{
 public:
    TrieNode(char c): _ch(c), end(false){}
    char _ch;
    bool end;
    unordered_map<char, TrieNode *> children;
};

class Trie
{
public:
    Trie(): root(new TrieNode('\0')) {}
    void Insert(string & word)
    {
        TrieNode * p = root;
        int i = 0;
        for(; i < word.size(); ++i)
        {
            if(p->children.find(word[i]) == end(p->children))
            {
                p->children[word[i]] = new TrieNode(word[i]);
            }
            p = p->children[word[i]];
        }
        p->end = true;
    }
    TrieNode * GetRoot() {return root;}
private:
    TrieNode *root;
};

class Solution 
{
public:
    vector<string> findWords(vector<vector<char>>& board, vector<string>& words) 
    {
        Trie tr;
        for(auto & e : words)
        {
            tr.Insert(e);
        }
        vector<string> res;
        string curr;
        int I = board.size();
        int J = board[0].size();
        for(int i = 0; i < I; ++i)
        {
            for(int j = 0; j < J; ++j)
            {
                rec(board, I, i, J, j, tr.GetRoot(), res, curr);
            }
        }
        return res;
    }
    
    void rec(vector<vector<char> > &board, int I, int i, int J, int j, TrieNode * p, vector<string> &res, string &curr)
    {
        if(p)
        {
            if(p->end)
            {
                res.push_back(curr);
                p->end = false;
            }
            if(i >= 0 && i < I && j >= 0 && j < J && p->children.find(board[i][j]) != end(p->children))
            {
                char old = board[i][j];
                p = p->children[old];
                curr.append(1, old);
                board[i][j] = 'X';
                rec(board, I, i + 1, J, j, p, res, curr);
                rec(board, I, i - 1, J, j, p, res, curr);
                rec(board, I, i, J, j + 1, p, res, curr);
                rec(board, I, i, J, j - 1, p, res, curr);
                board[i][j] = old;
                curr = curr.substr(0, curr.length() - 1);
            }
        }
    }
};
