Palindrome Partitioning 

class Solution {
public:
    bool isPalindrome(string &s, int i, int j)
    {
        for(; i <= j && s[i] == s[j]; ++i, --j) {  }
        return i > j;
    }
    vector<vector<string>> partition(string s) 
    {
        int N = s.size();
        vector<vector<vector<string> >> vt(N);
        
        for(int i = N - 1; i >= 0; --i)
        {
            for(int j = i; j < N; ++j)
            {
                if(isPalindrome(s, i, j))
                {
                    string sub = s.substr(i, j - i + 1);
                    if(j + 1 == N)
                    {
                        vector<string> t(1, sub);
                        vt[i].push_back(t);
                    }
                    else
                    {
                        for(auto & e : vt[j + 1])
                        {
                            vector<string> t(e);
                            t.insert(t.begin(), sub);
                            vt[i].push_back(t);
                        }                         
                    }
                }
            }
        }
        
        return vt[0];
    }
};
