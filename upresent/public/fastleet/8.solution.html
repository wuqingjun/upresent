String to Integer (atoi) 

class Solution 
{
public:
    int atoi(const char *str) 
    {
        long long res = 0;
        int flag = 1;
        while(*str == ' '){++str;}
        if(*str == '-'|| *str == '+') 
        {
            if(*str == '-') {flag = -1;}
            ++str;
        }
        for(;*str != '\0' && *str >= '0' && *str <= '9'; ++str)
        {
            res = res * 10 + *str - '0';
            if(flag * res >= INT_MAX) {return INT_MAX;}
            if(flag * res <= INT_MIN) {return INT_MIN;}
        }
        return flag * res;
    }
};
