Generate Parentheses 

class Solution {
public:
    vector<string> generateParenthesis(int n) 
    {
        vector<string> res;
        rec(0, 0, n, string(""), res);
        return res;
    }
    void rec(int i, int j, int n, string curr, vector<string> & res)
    {
        if(i == n && j == 0)
        {
            res.push_back(curr);
        }
        else if(i <= n && j >= 0)
        {
            rec(i, j - 1, n, curr + string(1, ')'), res);
            rec(i + 1, j + 1, n, curr + string(1, '('), res);
        }
    }
};
