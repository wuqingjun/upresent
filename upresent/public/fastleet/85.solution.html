Maximal Rectangle 

class Solution {
public:
    int maximalRectangle(vector<vector<char> > &matrix) 
    {
        int res = 0;
        if(matrix.size() > 0 && matrix[0].size() > 0)
        {
            int m = matrix.size(), n = matrix[0].size();
            vector<vector<pair<int, int> > > dp(m, vector<pair<int, int> > (n, {0, 0}));
            for(int i = 0; i < m; ++i)
            {
                for(int j = 0; j < n; ++j)
                {
                    if(matrix[i][j] == '1')
                    {
                        dp[i][j].first = j > 0 ? dp[i][j - 1].first + 1 : 1;
                        dp[i][j].second = i > 0 ? dp[i - 1][j].second + 1 : 1;
                        for(int a = i, w = dp[i][j].first; a >= 0; --a)
                        {
                            w = min(w, dp[a][j].first);
                            res = max(res, (i - a + 1) * w);
                        }
                        for(int b = j, h = dp[i][j].second; b >= 0; --b)
                        {
                            h = min(h, dp[i][b].second);
                            res = max(res, (j - b + 1) * h);
                        }                        
                    }
                }
            }            
        }
        return res;
    }
};
