Regular Expression Matching 

class Solution 
{
public:
    bool isMatch(const char *s, const char *p) 
    {
        int N = strlen(s), M = strlen(p);
        
        vector<bool> dp((N + 1) * (M + 1), false);
        for(int n = N; n >= 0; --n)
        {
            for(int m = M; m >= 0; --m)
            {
                if(n == N && m == M)
                {
                    dp[n * (M + 1) + m] = true;
                }
                else if(m + 1 < M && p[m + 1] == '*')
                {
                    dp[n * (M + 1) + m] = dp[n * (M + 1) + m + 1];
                }
                else
                {
                    int k = '*' == p[m] ? m - 1 : m;
                    dp[n * (M + 1) + m] = ((p[k] == s[n] || p[k] == '.') && (n < N && dp[(n + 1) * (M + 1) + k + 1]) )
                        || ('*' == p[m] && dp[n * (M + 1) + m + 1]);
                }
            }
        }
        return dp[0];
    }

};
