Swap Nodes in Pairs 

class Solution {
public:
    ListNode *swapPairs(ListNode *head) 
    {
        ListNode *nhead = head;
        for(ListNode *last = NULL, *nextnext = NULL; head && head->next; 
                    head->next = nextnext, last = head, head = nextnext)
        {
            nextnext = head->next->next, 
            head->next->next = head;
            if(last) { last->next = head->next; }
            else { nhead = head->next; }
        }
        return nhead;
    }
};
