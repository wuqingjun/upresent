Permutations II 

class Solution {
public:
    vector<vector<int> > permuteUnique(vector<int> &num) 
    {
        sort(begin(num), end(num));
        vector<vector<int> > res;
        set<vector<int> > s;
        rec(num, 0, s);
        for(auto & e : s) {res.push_back(e);}
        return res;
    }
    void rec(vector<int> &num, int i, set<vector<int> > &res)
    {
        if(i == num.size())
        {
            res.insert(num);
        }
        for(int j = i; j < num.size(); ++j)
        {
            if(i == j || num[i] != num[j])
            {
                swap(num[i], num[j]);
                rec(num, i + 1, res);
                swap(num[i], num[j]);
            }
        }
    }
};
