Divide Two Integers 

class Solution {
public:
    int divide(int dividend, int divisor)
    {
        int sign = (dividend & 1 << 31) ^ (divisor & 1 << 31) ? -1 : 1; 
        long long dd = abs((long long) (dividend)), ds = abs((long long) (divisor));
        int shift = 0, res = 0;
        for (;  dd >= (ds << 1); ds <<= 1, ++shift) {}
        if(shift == 31) { return sign > 0 ? INT_MAX : INT_MIN; }
        for (; shift >= 0; --shift, ds >>= 1)
        {
            if (dd >= ds) { res += 1 << shift, dd -= ds; }
        }
        return res * sign;
    }
};
