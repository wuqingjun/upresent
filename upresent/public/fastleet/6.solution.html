ZigZag Conversion 

class Solution 
{
public:
	string convert(string s, int nRows)
	{
		string res;
		for (int i = 0; i < nRows; ++i)
		{
		    int dist[2] = { 2 * nRows - 2 - i * 2, 2 * i };
			for (int j = i, k = 0, prev = -1; j < s.size(); ++k)
			{
				if (j != prev) { res.append(1, s[j]); }
				prev = j, j += nRows == 1 ? 1 : dist[k % 2];
			}
		}
		return res;
	}
};
