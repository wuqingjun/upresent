Letter Combinations of a Phone Number 

class Solution {
public:
    vector<string> letterCombinations(string digits) 
    {
        string dict[] = {"", " ", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"};
        vector<string> res;
        rec(digits, 0, dict, res, string(""));
        return res;
    }
    void rec(string & digits, int i, string dict[], vector<string> &res, string curr)
    {
        if(i < digits.size())
        {
            for(int j = 0; j < dict[digits[i] - '0'].size(); ++j)
            {
                rec(digits, i + 1, dict, res, curr + string(1, dict[digits[i] - '0'][j]));
            }
        }
        else { res.push_back(curr); }
    }
};
