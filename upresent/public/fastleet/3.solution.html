Longest Substring Without Repeating Characters 

class Solution 
{
public:
        int lengthOfLongestSubstring(string s) 
        {
        unordered_map<char, int> mp;
        int maxlen = 0;
        for(int i = 0, startidx = 0; i < s.length(); mp[s[i]] = i, ++i)
        {
        if(mp.find(s[i]) != end(mp) && mp[s[i]] >= startidx)
        {
                startidx = mp[s[i]] + 1;
        }
            maxlen = max(maxlen, i - startidx + 1);
        }
        return maxlen;
        }
};
