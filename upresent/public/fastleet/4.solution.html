Median of Two Sorted Arrays 

class Solution 
{
public:
    double findMedianSortedArrays(int A[], int m, int B[], int n) 
    {
        if((m + n) & 0x01) { return findKth(A, m, B, n, (m + n) / 2 + 1);}
        else {return double(findKth(A, m, B, n, (m + n) / 2) + findKth(A, m, B, n, (m + n) / 2 + 1)) / 2; }
    }
    int findKth(int A[], int m, int B[], int n, int k)
    {
        if(m > n) { return findKth(B, n, A, m, k); }
        if(m == 0) { return B[k - 1]; }
        else if(k == 1) {return min(A[0], B[0]); }
        int a = min(k / 2, m), b = k - a;
        if(A[a - 1] < B[b - 1]) { return findKth(A + a, m - a, B, b, k - a); }
        else if(A[a - 1] > B[b - 1]) {return findKth(A, m, B + b, n + a - k, k - b);}
        else {return A[a - 1];}
    }
};
