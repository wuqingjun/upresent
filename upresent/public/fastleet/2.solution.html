Add Two Numbers 

class Solution 
{
public:
	ListNode *addTwoNumbers(ListNode *l1, ListNode *l2) 
	{
		int sum = 0;
		ListNode * head = NULL;
		for(ListNode * curr = NULL, * temp = NULL; l1 || l2 || sum != 0; sum /= 10, curr = temp)
		{
			if(l1)
			{
				sum += l1->val;
				l1 = l1->next;
			}
			if(l2)
			{
				sum += l2->val;
				l2 = l2->next;
			}
			temp = new ListNode(sum % 10);
			if(curr)
			{
				curr->next = temp;
			}
			else
			{
				head = temp;
			}
		}
		return head;
	}
};
