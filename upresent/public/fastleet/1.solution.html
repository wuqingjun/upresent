Two Sum 

class Solution {
public:
    vector<int> twoSum(vector<int> &numbers, int target) 
    {
        unordered_map<int, vector<int>> mp;
        for(int i = 0; i < numbers.size(); ++i)
        {
            mp[numbers[i]].push_back(i + 1);
        }
        vector<int> res;
        for(auto & e : mp)
        {
            auto f = target - e.first;
            if(e.first != f && mp.find(f) != end(mp))
            {
                res.push_back(min(e.second[0], mp[f][0])), res.push_back(max(e.second[0], mp[f][0]));
            }
            else if(e.first == f && e.second.size() > 1)
            {
                res.push_back(min(e.second[0], e.second[1])), res.push_back(max(e.second[0], e.second[1]));
            }
        }
        return res;
    }
};
