Valid Palindrome 

class Solution {
private:
    char lowercase(char c) 
    {
        if(c >= 'A' && c <= 'Z')
        {
            return 'a' + c - 'A';
        }
        return c;
    }
    bool isAlphanumeric(char c) {return (c >= 'a' && c <'z') || (c >= '0' && c <= '9'); }
public:
    bool isPalindrome(string s) 
    {
        int p[] = {0, (int)s.size() - 1};
        
        while(p[0] < p[1])
        {
            char ch[] = {lowercase(s[p[0]]), lowercase(s[p[1]])};
            bool cont = true;
            for(int i = 0; cont && i < 2; ++i)
            {
                if(!isAlphanumeric(ch[i]))
                {
                    p[i] += i ? -1 : 1;
                    cont = false;
                }
            }
            if(cont)
            {
                if(ch[0] != ch[1])
                {
                    return false;
                }
                ++p[0], --p[1];
            }
        }
        return true;
    }
};
