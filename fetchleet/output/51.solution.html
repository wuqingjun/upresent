N-Queens 

class Solution {
public:
    vector<vector<string> > solveNQueens(int n) 
    {
        string init(n, '.');
        vector<string> curr(n, init);
        vector<vector<string> > res;
        vector<int> cols(n, 0);
        vector<unordered_set<int> > s(2);
        rec(n, 0, res, s, cols, curr);
        return res;
    }
    
    void rec(int n, int i, vector<vector<string> > &res, vector<unordered_set<int> > &s, vector<int> &cols, vector<string> &curr)
    {
        if(i == n) { res.push_back(curr); }
        for(int j = 0; i < n && j < n; ++j)
        {
            if(cols[j] == 0 && s[0].find(i + j + 2) == end(s[0]) && s[1].find(i - j) == end(s[1]))
            {
                cols[j] = 1, curr[i][j] = 'Q', s[0].insert(i + j + 2), s[1].insert(i - j);
                rec(n, i + 1, res, s, cols, curr);
                s[0].erase(i + j + 2), s[1].erase(i - j), curr[i][j] = '.', cols[j] = 0;
            }
        }
    }
};
