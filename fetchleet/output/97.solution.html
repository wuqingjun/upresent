Interleaving String 

class Solution {
public:
    bool isInterleave(string s1, string s2, string s3)
    {
        int l1 = s1.size(), l2 = s2.size(), l3 = s3.size();
        if (l1 + l2 == l3)
        {
            vector<vector<int>> dp(l1 + 1, vector<int>(l2 + 1, 0));
            for (int i = 0; i <= l3; ++i)
            {
                for (int j = 0, k = i - j; j <= min(i, l1); ++j, k = i - j)
                {
                    if (k <= l2)
                    {
                        dp[j][k] |= (i && j && s3[i - 1] == s1[j - 1] && dp[j - 1][k])
                            || (i && k && s3[i - 1] == s2[k - 1] && dp[j][k - 1])
                            || (!k && !j);
                    }
                }
            }
            return dp[l1][l2];
        }
        return false;
    }
};
