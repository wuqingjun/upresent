3Sum Closest 

class Solution {
public:
    int threeSumClosest(vector<int> &num, int target) 
    {
        int curr = INT_MAX, res = 0;
        sort(begin(num), end(num));
        for(int i = 0, n = num.size(); i < n - 2; ++i)
        {
            for(int s = i + 1, e = n - 1; s < e; )
            {
                int sum = num[i] + num[s] + num[e];
                if(abs(sum - target) < curr)
                {
                    curr = abs(sum - target), res = sum;
                }
                if(sum > target) { --e; }
                else if(sum < target) { ++s; }
                else { return sum; }
            }
        }
        return res;
    }
};
