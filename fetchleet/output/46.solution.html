Permutations 

class Solution {
public:
    vector<vector<int> > permute(vector<int> &num) 
    {
        vector<vector<int> > res;
        rec(num, 0, res);
        return res;
    }
    
    void rec(vector<int> &num, int i, vector<vector<int> > &res)
    {
        if(i == num.size())
        {
            res.push_back(num);
        }
        for(int j = i; j < num.size(); ++j)
        {
            swap(num[i], num[j]);
            rec(num, i + 1, res);
            swap(num[i], num[j]);
        }
    }
};
