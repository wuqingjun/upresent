Sort List 

class Solution {
public:
    ListNode *sortList(ListNode *head) 
    {
        if(head && head->next)
        {
            ListNode *h2 = divide(head);
            ListNode *h1 = sortList(head);
            h2 = sortList(h2);
            head = merge(h1, h2);
        }
        return head;
    }
    
    ListNode *divide(ListNode *h)
    {
        ListNode *p1 = h, *p2 = h->next, *h2 = NULL;
        for(int i = 0; p2 && p2->next; i = (i + 1) % 2, p2 = p2->next)
        {
            if(i) 
            {
                p1 = p1->next;
            }
        }
        h2 = p1->next, p1->next = NULL;
        return h2;
    }
    
    ListNode * merge(ListNode * h1, ListNode *h2)
    {
        ListNode *h = NULL, *last = NULL;
        while(h1 || h2)
        {
            ListNode *& r = (!h2 || (h1 && h2 && h1->val < h2->val)) ? h1 : h2;
            if(last)
            {
                last->next = r;
            }
            else
            {
                h = r;
            }
            last = r, r = r->next;
        }
        return h;
    }
};
