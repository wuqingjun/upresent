Trapping Rain Water 

class Solution {
public:
    int trap(int A[], int n) 
    {
        vector<int> lr(n, 0), rl(n, 0);
        for(int i = 0; i < n; ++i)
        {
            lr[i] = i > 0 ? max(A[i], lr[i - 1]) : A[i];
            rl[n - 1 - i] = i > 0 ? max(A[n - 1 - i], rl[n - i]) : A[n - 1 - i];
        }
        int sum = 0;
        for(int i = 0; i < n; ++i)
        {
            sum += max(min(lr[i], rl[i]) - A[i], 0);
        }
        return sum;
    }
};
