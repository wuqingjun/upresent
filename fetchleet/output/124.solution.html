Binary Tree Maximum Path Sum 

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int maxPathSum(TreeNode *root) 
    {
        int m = INT_MIN;
        rec(root, m);
        return m;
    }
    
    int rec(TreeNode * root, int &m)
    {
        if(root)
        {
            int l = rec(root->left, m);
            int r = rec(root->right, m);
            int res = root->val;
            res = max(res, root->val + l);
            res = max(res, root->val + r);
            m = max(res, m);
            m = max(m, root->val + l + r);
            return res;
        }
        return 0;
    }
};
