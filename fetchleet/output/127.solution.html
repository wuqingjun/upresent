Word Ladder 

class Solution {
public:
    int ladderLength(string start, string end, unordered_set<string> &dict) 
    {
        queue<string> que[2];
        que[0].push(start);
        unordered_set<string> visited;
        visited.insert(start);
        for(int i = 0; !que[i % 2].empty() ;++i)
        {
            for(string t; !que[i % 2].empty(); que[i % 2].pop())
            {
                t = que[i % 2].front();
                if(t == end)
                {
                    return i + 1;
                }
                for(int n = 0; n < t.size(); ++n)
                {
                    char old = t[n];
                    for(int k = 0; k < 26; ++k)
                    {
                        t[n] = 'a' + k;
                        if(dict.find(t) != dict.end() && visited.find(t) == visited.end())
                        {
                            que[(i + 1) % 2].push(t);
                            visited.insert(t);
                        }
                    }
                    t[n] = old;
                }
            }
        }
        return 0;
    }
};
