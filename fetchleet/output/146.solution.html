LRU Cache 

struct Node
{
    Node(int k, int v) : key(k), val(v), prev(NULL), next(NULL) {}
    int key;
    int val;
    Node *prev;
    Node *next;
};

class DLL
{
public:
    Node * head;
    Node * tail;
public:
    DLL()
    {
        head = new Node(-1, -1);
        tail = new Node(-1, -1);
        head->next = tail;
        tail->prev = head;
    }
    void append(Node * n)
    {
        tail->prev->next = n;
        n->prev = tail->prev;
        n->next = tail;
        tail->prev = n;
    }
    void remove(Node * n)
    {
        n->prev->next = n ->next;
        n->next->prev = n->prev;
    }
};

class LRUCache
{
private:
    int Size;
    int Count;
    unordered_map<int, Node *> mp;
    DLL dll;
public:
    LRUCache(int capacity) : Size(capacity), Count(0)
    {
        
    }
    
    int get(int key) 
    {
        if(mp.find(key) != end(mp))
        {
            dll.remove(mp[key]);
            dll.append(mp[key]);
            return mp[key]->val;
        }
        return -1;
    }
    
    void set(int key, int value) 
    {
        if(mp.find(key) != end(mp))
        {
            mp[key]->val = value;
            dll.remove(mp[key]);
            dll.append(mp[key]);
        }
        else
        {
            auto * p = new Node(key, value);
            if(Count == Size)
            {
                auto itr = mp.find(dll.head->next->key);
                mp.erase(itr);
                dll.remove(dll.head->next);
                dll.append(p);
                mp[key] = p;
            }
            else
            {
                mp[key] = p;
                dll.append(p);
                ++Count;
            }
        }
    }
};
