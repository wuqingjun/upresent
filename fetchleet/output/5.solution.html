Longest Palindromic Substring 

class Solution {
public:
    string longestPalindrome(string s)
    {
        string t("#"), res;
        for (int i = 0; i < s.size(); ++i)
        {
            t.append(1, s[i]), t.append(1, '#');
        }
        int id = 0, mx = 0, ml = 0, center = 0;
        vector<int> p(t.size(), 0);
        for (int i = 0; i < t.size(); ++i)
        {
            p[i] = i < mx ? min(p[2 * id - i], mx - i) : 1;
            while (i + p[i] < t.size() && i >= p[i] && t[i + p[i]] == t[i - p[i]]) { ++p[i]; }
            if (i + p[i] - 1 > mx)
            {
                id = i;
                mx = i + p[i] - 1;
            }
            if (p[i] > ml)
            {
                ml = p[i];
                center = i;
            }
        }
        for (int i = center - p[center] + 1; i <= center + p[center] - 1; ++i)
        {
            if ('#' != t[i])
            {
                res.append(1, t[i]);
            }
        }
        return res;
    }
};
