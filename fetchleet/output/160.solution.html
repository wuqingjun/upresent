Intersection of Two Linked Lists 

class Solution {
public:
    ListNode *getIntersectionNode(ListNode *headA, ListNode *headB) 
    {
        ListNode *res = NULL;
        int mask = 0X80000000;
        for(ListNode *next = NULL, *pa = headA; NULL != pa; pa = next)
        {
            next = pa->next;
            pa->next = (ListNode *)(int(pa->next) |mask);
        }
        for(ListNode *pb = headB; NULL != pb && NULL == res; pb = pb->next)
        {
            if(mask & int(pb->next)) { res = pb; }
        }
        for(ListNode *pa = headA; NULL != pa; pa->next = (ListNode * )(int(pa->next) ^ mask), pa = pa->next) { }
        return res;
    }
};
