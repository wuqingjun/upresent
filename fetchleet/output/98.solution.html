Validate Binary Search Tree 

/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isValidBST(TreeNode *root) 
    {
        TreeNode *prev = NULL;
        return rec(root, prev);
    }
    bool rec(TreeNode *root, TreeNode *&prev)
    {
        bool res = true;
        if(root)
        {
            res = rec(root->left, prev) && (!prev || prev->val < root->val);
            prev = root;
            res &= rec(root->right, prev);
        }
        return res;
    }
};
