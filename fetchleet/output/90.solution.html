class Solution {
public:
    vector<vector<int> > subsetsWithDup(vector<int> &S) 
    {
        vector<vector<int> > res;
        sort(begin(S), end(S));
        vector<int> r;
        rec(S, 0, res, r);
        return res;
    }
    
    void rec(vector<int> & S, int i, vector<vector<int> > &res, vector<int> & r)
    {
        if(i == S.size())
        {
            res.push_back(r);
        }
        else
        {
            r.push_back(S[i]);
            rec(S, i + 1, res, r);
            r.pop_back();
            for(++i; i < S.size() && S[i] == S[i - 1]; ++i) {  }
            rec(S, i, res, r);
        }
    }
};
