First Missing Positive 

class Solution {
public:
    int firstMissingPositive(int A[], int n) 
    {
        for(int i = 0; i < n; ++i)
        {
            for(int t = A[i]; t > 0 && t <= n && t!= A[t - 1]; swap(t, A[t - 1])) { }
        }
        int i = 0;
        for(; i < n && A[i] == i + 1; ++i) {   }
        return i + 1;
    }
};
