Recover Binary Search Tree 

class Solution {
public:
    void recoverTree(TreeNode *root) 
    {
        TreeNode *prev = NULL;
        vector<TreeNode *> nodes;
        rec(root, prev, nodes);
        swap(nodes[0]->val, nodes[1]->val);
    }
    void rec(TreeNode *p, TreeNode *&prev, vector<TreeNode *> &nodes)
    {
        if(p)
        {
            rec(p->left, prev, nodes);
            if(prev && prev->val > p->val)
            {
                if(nodes.size() == 0)
                {
                    nodes.push_back(prev), nodes.push_back(p);
                }
                else
                {
                    nodes[1] = p;
                }
            }
            prev = p;
            rec(p->right, prev, nodes);
        }
    }
};
