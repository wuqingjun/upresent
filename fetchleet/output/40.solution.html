Combination Sum II 

class Solution {
public:
    vector<vector<int> > combinationSum2(vector<int> &candidates, int target) 
    {
        sort(begin(candidates), end(candidates));
        vector< vector<int> > res;
        set<vector<int> > s;
        map<vector<int>, int> mp;
        rec(candidates, candidates.size() - 1, target, s, mp);
        for(auto & e : s) {res.push_back(e);}
        return res;
    }
    void rec(vector<int> &candidates, int i, int t, set<vector<int> > & s, map<vector<int>, int> &mp)
    {
        if(i == -1)
        {
            mp[vector<int>()] = 0;
        }
        else
        {
            rec(candidates, i - 1, t, s, mp);
            map<vector<int>, int> temp(mp);
            for(auto &e: temp)
            {
                vector<int> v(e.first);
                v.push_back(candidates[i]);
                if(e.second +  candidates[i] == t)
                {
                    s.insert(v);
                }
                else if(e.second + candidates[i]< t)
                {
                    mp[v] = e.second + candidates[i];
                }
            }
        }
    }    
};
