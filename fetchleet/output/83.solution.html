Remove Duplicates from Sorted List 

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *deleteDuplicates(ListNode *head) 
    {
        for(ListNode * p = head, * prev = NULL; p; p = p->next)
        {
            if(!prev || prev->val != p->val)
            {
                prev = p;
            }
            else
            {
                prev->next = p->next;
            }
        }
        return head;
    }
};
