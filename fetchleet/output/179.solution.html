Largest Number 

class Solution {
public:
    string largestNumber(vector<int>& nums) 
    {
        sort(begin(nums), end(nums), [](int s1, int s2)
        {
            string n1 = to_string(s1) + to_string(s2);
            string n2 = to_string(s2) + to_string(s1);
            return n1 > n2;
        });
        string res;
        for(auto & e : nums)
        {
            res += to_string(e);
        }
        return count(res.begin(), res.end(), '0') == res.size() ? "0" : res;
    }
    
};
