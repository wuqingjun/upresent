Restore IP Addresses 

class Solution {
public:
    vector<string> restoreIpAddresses(string s) 
    {
        return rec(s, 4);
    }
    vector<string> rec(string s, int n)
    {
        vector<string> res;
        if(n == 0 && s.empty())
        {
            res.push_back(s);
        }
        for(int i = 1, sum = 0; i <= min(3, (int)s.length()) && n > 0; ++i)
        {
            sum = 10 * sum + s[i - 1] - '0';
            auto sub = s.substr(0, i);
            if(i == 1 || (sub[0] != '0' && sum >= 0 && sum < 256) )
            {
                vector<string> t = rec(s.substr(i), n - 1);
                for(auto & e : t)
                {
                    res.push_back(sub + (e.empty() ? "" : ".") + e);
                }
            }
        }
        return res;
    }
};
