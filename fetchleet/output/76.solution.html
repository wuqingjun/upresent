Minimum Window Substring 

class Solution {
public:
    string minWindow(string S, string T) 
    {
        unordered_map<char, int> s, t;
        for(auto & e : T) { t[e]++; }
        int count = 0, finalstart = -1, start = -1, len = INT_MAX;
        for(int i = 0; i < S.size(); ++i)
        {
            if(t.find(S[i]) != end(t))
            {
                if(-1 == start) { start = i; }
                if(++s[S[i]] == t[S[i]]) { ++count; }
                for(; count == t.size(); ++start)
                {
                    if(i - start + 1 < len)
                    {
                        len = i - start + 1, finalstart = start;
                    }
                    if(s.find(S[start]) != end(s) && --s[S[start]] < t[S[start]])
                    {
                        --count;
                    }
                }
            }
        }
        return finalstart >= 0 ? S.substr(finalstart, len) : "";
    }
};
