Candy 

class Solution {
public:
    int candy(vector<int> &ratings) 
    {
        int sum = 0, N = ratings.size();
        for(int i = 0, s = 0; i < N; ++i)
        {
            if(i == N - 1 || ratings[i] == ratings[i + 1])
            {
                sum += helper(ratings, s, i);
                s = i + 1;
            }
        }
        return sum;
    }
        
    int helper(vector<int> &ratings, int s, int e)
    {
        if (s == e) return 1;
        int sum = 0;
        int i = s;
        while (i < e)
        {
            int l = i, h = i;
            while (i < e && ratings[i + 1] > ratings[i])
            {
                h = i + 1;
                ++i;
            }
            int r = i;
            while (i < e && ratings[i + 1] < ratings[i])
            {
                r = i + 1;
                ++i;
            }
            sum += (2 + h - l) * (h - l + 1) / 2 + (r - h + 2) * (r - h + 1) /2 - min(h - l + 1, r - h + 1);
            if (i > s && i < e)
            {
                sum -= 1;
            }
        }
        return sum;
    }
};
